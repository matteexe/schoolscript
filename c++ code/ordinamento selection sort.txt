/*CAMBIARE IL NOME DEL VETTORE*/

void ordinamento(int v[DMAX], int n)
{
    int i, j, tmp;
    bool check = false;

    for (i = 0; i < n - 1; i++)
    {
        for (j = i + 1; j < n; j++)
        {
            if (v[i] > v[j])
            {
                tmp = v[i];
                v[i] = v[j];
                v[j] = tmp;
            }
        }
    }
}
